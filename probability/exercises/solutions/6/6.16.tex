    Let 
    \begin{align}
    X \in \{1,2,3,4,5,6\}
    \end{align}
    $\because$
    \begin{align}
    \pr{X=i}=\begin{cases}
        \frac{1}{6}& 1 \le i \le 6\\
        0 & otherwise
        \end{cases},
    \end{align}
    %
    \begin{align}
        \pr{A} &={\sum_{i=2,4,6}\pr{X=i}}=\frac{1}{2} \\
        \pr{B}  &= {\sum_{i=1}^{3}\pr{X=i}}=\frac{1}{2} \\
        \pr{AB} &= \pr{X=2}=\frac{1}{6}\\
         \pr{A}\times \pr{B} &= \frac{1}{4}
    \end{align}
    %
    $\because$
    \begin{align}
        \pr{A}\times \pr{B} \neq \pr{AB},
    \end{align}
    $ A$ and $B$ are not independent.
